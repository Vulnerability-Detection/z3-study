安装

Linux安装教程

git clone https://github.com/Z3Prover/z3.git
cd z3
python scripts/mk_make.py
cd build
make
make install

window10安装z3-solver python版本

  • 注意pip安装的是z3-solver，而不是z3。

测试-python文件运行

  • 输入xx.python
# 示例1
from z3 import *

if __name__ == "__main__":
    x, y, z = Ints('x y z')
    s = Solver()
    s.add(2 * x + 3 * y + z == 6)
    s.add(x - y + 2 * z == -1)
    s.add(x + 2 * y - z == 5)
    # 解方程=>  x = 2, y = 1, z = -1
    # 2x + 3y + z = 6
    # x - y + 2z = -1
    # x + 2y - z = 5
    print(s.check())  # sat
    print(s.model())  # [z = -1, y = 1, x = 2]
  • 输出
sat
[z = -1, y = 1, x = 2]
  • 输入：xx.python
# 示例2
from z3 import *

if __name__ == "__main__":
    x = Int('x')
    s = Solver()
    a = 65537
    b = 64834
    c = 41958
    s.add(x > 0)
    s.add(x % a == b)
    s.add(x % b == c)

    print(s.check())  # sat
    print(s.model())  # [x = 227609298]
  • 输出
sat
[x = 227609298]
